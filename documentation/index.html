Длительность видео в таблице

  • Данное приложение решает следующие задачи:
  • В поля формы вводиться название видео, сколько оно длится часов, минут и секунд.
  • После нажатия кнопки "Добавить запись в таблицу", происходит проверка полей на правильность введенных данных.
  • Если данные в поле введены неверно, то поле меняет стиль. Его рамка становиться красной, а под инпутом появляется сообщение, которое объясняет, что не так с данными введенными в поле.
  • Проверка поля происходит на пустую строку, на то, что поле является отрицательным или не целым числом, и на то, что оно больше 59.
  • Если данные в поле введены верно, то поле меняет стиль. Его рамка становится зеленой.
  • Если все данные введенне пользователем верны, запись добавляется в таблицу, а поля очищаются от введенных в них данных.
  • Приложение позволяет как заполнить таблицу с нуля, так задать их с помощью массива. При этом если пользователь продолжит добавлять записи в таблицу, они будут добавлены в уже существующий массив.

  •               function validateFormData(string, emptyLine, notANumber, range, start, end)
              {
                let errors = [];
                let errorEmptyLine = false;
                let errorNotANumber = false;
                let errorRange = false;

                if (emptyLine) {
                  errorEmptyLine = checkData(string, checkForEmptyLine);

                  if (errorEmptyLine) {
                    errors.push(true);
                    errors.push(`${EMPTY_STRING}`);
                  }
                }

                if (notANumber) {
                  errorNotANumber = checkData(string, checkNotANumber);

                  if (errorNotANumber) {
                    if (errors.length === 0) {
                      errors.push(true);
                      errors.push(`${NOT_A_NUMBER}`);
                    } else {
                      errors[1] = `${NOT_A_NUMBER}`;
                    }
                  }
                }

                if (range) {
                  errorRange = checkData(string, checkRange
                    .bind(
                      checkRange,
                      string,
                      start,
                      end
                    )
                  );

                  if (errorRange) {
                    if (errors.length === 0) {
                      errors.push(true);
                    errors.push(`${OUT_OF_RANGE}`);
                    }   else {
                    errors[1] = `${OUT_OF_RANGE}`;
                    }
                  }
                }

                return errors;
              }
